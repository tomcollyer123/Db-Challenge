SELECT C.CATEGORY_ID, C.NAME, FC.CATEGORY_ID
FROM CATEGORY AS C
JOIN FILM_CATEGORY AS FC
on C.CATEGORY_ID = FC.CATEGORY_ID
GROUP BY C.CATEGORY_ID, C.NAME
ORDER BY COUNT(FC.FILM_ID) DESC LIMIT 1;
